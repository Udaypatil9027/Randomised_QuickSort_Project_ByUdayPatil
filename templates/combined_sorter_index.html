Understanding Randomized Quick Sort

Why Use Randomized Quick Sort in CRM Systems?

Customer Relationship Management (CRM) systems often need to sort customer data. When this data arrives partially sorted (by date, previous sorting, etc.), traditional Quick Sort can become very slow.

Randomized Quick Sort solves this by randomly selecting pivot elements, ensuring that the algorithm performs well on average, regardless of the initial order of data.

💡 Key Insight: By making the pivot selection random, we prevent the worst-case scenario where the data is already sorted, which would make standard Quick Sort very inefficient.

How Randomized Quick Sort Works

1 Select a Random Pivot: Instead of always picking the first or last element, we randomly select an element from the current segment of the array.
2 Partition the Array: Rearrange the array so that all elements smaller than the pivot come before it, and all elements greater come after it.
3 Recursively Sort: Apply the same process to the sub-arrays on both sides of the pivot.
4 Repeat: Continue this process until the entire array is sorted.

Time Complexity Comparison

Algorithm Best Case Average Case Worst Case Space Complexity
Standard Quick Sort O(n log n) O(n log n) O(n²) - when data is sorted O(log n)
Randomized Quick Sort O(n log n) O(n log n) O(n log n) - with high probability O(log n)

import random

def randomized_quick_sort(data, low, high):
    # Base case: if the segment has more than one element
    if low < high:
        # Get a random partition index
        p = randomized_partition(data, low, high)
        # Recursively sort elements before and after partition
        randomized_quick_sort(data, low, p - 1)
        randomized_quick_sort(data, p + 1, high)

def randomized_partition(data, low, high):
    # Select a random index between low and high
    random_index = random.randint(low, high)
    # Swap the random element with the last element
    data[random_index], data[high] = data[high], data[random_index]
    # Now partition using the last element as pivot
    return partition(data, low, high)

def partition(data, low, high):
    # Use the last element as pivot (after randomization)
    pivot = data[high].lower()  # Case-insensitive comparison
    i = low - 1  # Index of smaller element
    
    for j in range(low, high):
        # If current element is smaller than or equal to pivot
        if data[j].lower() <= pivot:
            i += 1
            # Swap elements at i and j
            data[i], data[j] = data[j], data[i]
    
    # Place the pivot in its correct position
    data[i + 1], data[high] = data[high], data[i + 1]
    return i + 1
            
import java.util.Random;

public class RandomizedQuickSort {
    
    public static void randomizedQuickSort(String[] data, int low, int high) {
        // Base case
        if (low < high) {
            // Get random partition index
            int p = randomizedPartition(data, low, high);
            // Recursively sort elements before and after partition
            randomizedQuickSort(data, low, p - 1);
            randomizedQuickSort(data, p + 1, high);
        }
    }
    
    private static int randomizedPartition(String[] data, int low, int high) {
        // Create random number generator
        Random rand = new Random();
        // Generate random index between low and high
        int randomIndex = low + rand.nextInt(high - low + 1);
        
        // Swap the random element with the last element
        String temp = data[randomIndex];
        data[randomIndex] = data[high];
        data[high] = temp;
        
        // Now call the standard partition method
        return partition(data, low, high);
    }
    
    private static int partition(String[] data, int low, int high) {
        String pivot = data[high].toLowerCase(); // Case-insensitive comparison
        int i = low - 1; // Index of smaller element
        
        for (int j = low; j < high; j++) {
            // If current element is smaller than or equal to pivot
            if (data[j].toLowerCase().compareTo(pivot) <= 0) {
                i++;
                // Swap data[i] and data[j]
                String temp = data[i];
                data[i] = data[j];
                data[j] = temp;
            }
        }
        
        // Swap data[i+1] and data[high] (or pivot)
        String temp = data[i + 1];
        data[i + 1] = data[high];
        data[high] = temp;
        
        return i + 1;
    }
}
            
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#include <ctype.h>

// Function to swap two strings
void swap(char **a, char **b) {
    char *temp = *a;
    *a = *b;
    *b = temp;
}

// Convert to lowercase for case-insensitive comparison
void toLowerCase(char *str) {
    for (int i = 0; str[i]; i++) {
        str[i] = tolower((unsigned char)str[i]);
    }
}

// Partition the array around a pivot
int partition(char **arr, int low, int high) {
    // Create copy for case-insensitive comparison
    char *pivot = malloc(strlen(arr[high]) + 1);
    strcpy(pivot, arr[high]);
    toLowerCase(pivot);
    
    int i = low - 1;

    for (int j = low; j < high; j++) {
        // Create copy for comparison
        char *current = malloc(strlen(arr[j]) + 1);
        strcpy(current, arr[j]);
        toLowerCase(current);
        
        // Compare case-insensitively
        if (strcmp(current, pivot) <= 0) {
            i++;
            swap(&arr[i], &arr[j]);
        }
        free(current);
    }
    
    swap(&arr[i + 1], &arr[high]);
    free(pivot);
    return i + 1;
}

// Randomized partition - core of the algorithm
int randomizedPartition(char **arr, int low, int high) {
    // Seed random number generator
    srand(time(NULL));
    // Generate random index between low and high
    int randomIndex = low + rand() % (high - low + 1);
    
    // Swap random element with last element
    swap(&arr[randomIndex], &arr[high]);
    
    // Call standard partition
    return partition(arr, low, high);
}

// Main Randomized Quick Sort function
void randomizedQuickSort(char **arr, int low, int high) {
    if (low < high) {
        // Get random partition index
        int pi = randomizedPartition(arr, low, high);
        
        // Recursively sort both partitions
        randomizedQuickSort(arr, low, pi - 1);
        randomizedQuickSort(arr, pi + 1, high);
    }
}
